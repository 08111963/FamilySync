Crea lo schema completo con queste tabelle:

1. **users** - Utenti dell'app
   - id (UUID, primary key)
   - email (VARCHAR 255, unique, not null)
   - password_hash (VARCHAR 255, not null)
   - name (VARCHAR 255)
   - avatar_url (TEXT)
   - created_at (TIMESTAMP, default NOW())
   - updated_at (TIMESTAMP, default NOW())

2. **families** - Gruppi familiari
   - id (UUID, primary key)
   - name (VARCHAR 255, not null)
   - avatar_url (TEXT)
   - color_theme (VARCHAR 7, default '#4A90E2')
   - subscription_status (ENUM: 'free', 'premium', default 'free')
   - stripe_customer_id (VARCHAR 255)
   - stripe_subscription_id (VARCHAR 255)
   - subscription_current_period_end (TIMESTAMP)
   - created_at (TIMESTAMP, default NOW())
   - updated_at (TIMESTAMP, default NOW())

3. **family_members** - Membri di una famiglia
   - id (UUID, primary key)
   - family_id (UUID, foreign key → families.id, ON DELETE CASCADE)
   - user_id (UUID, foreign key → users.id, ON DELETE CASCADE)
   - role (ENUM: 'admin', 'adult', 'teen', 'child')
   - nickname (VARCHAR 100) -- "Mamma", "Papà", etc.
   - color (VARCHAR 7) -- colore personale per calendario
   - joined_at (TIMESTAMP, default NOW())
   - UNIQUE(family_id, user_id)

4. **family_invites** - Inviti pending
   - id (UUID, primary key)
   - family_id (UUID, foreign key → families.id, ON DELETE CASCADE)
   - token (VARCHAR 255, unique, not null)
   - invited_by (UUID, foreign key → users.id)
   - role (ENUM: 'adult', 'teen', 'child')
   - expires_at (TIMESTAMP, not null)
   - accepted_at (TIMESTAMP)
   - created_at (TIMESTAMP, default NOW())

5. **calendar_events** - Eventi calendario
   - id (UUID, primary key)
   - family_id (UUID, foreign key → families.id, ON DELETE CASCADE)
   - title (VARCHAR 255, not null)
   - description (TEXT)
   - start_time (TIMESTAMP, not null)
   - end_time (TIMESTAMP, not null)
   - all_day (BOOLEAN, default false)
   - category (VARCHAR 50) -- 'work', 'school', 'sport', 'health', 'social', 'other'
   - location (VARCHAR 255)
   - recurrence_rule (TEXT) -- formato iCal RRULE
   - created_by (UUID, foreign key → users.id)
   - created_at (TIMESTAMP, default NOW())
   - updated_at (TIMESTAMP, default NOW())

6. **event_assignees** - Assegnazioni eventi
   - id (UUID, primary key)
   - event_id (UUID, foreign key → calendar_events.id, ON DELETE CASCADE)
   - user_id (UUID, foreign key → users.id, ON DELETE CASCADE)
   - UNIQUE(event_id, user_id)

7. **event_reminders** - Reminder eventi
   - id (UUID, primary key)
   - event_id (UUID, foreign key → calendar_events.id, ON DELETE CASCADE)
   - user_id (UUID, foreign key → users.id, ON DELETE CASCADE)
   - reminder_time (TIMESTAMP, not null)
   - sent (BOOLEAN, default false)

8. **shopping_lists** - Liste della spesa
   - id (UUID, primary key)
   - family_id (UUID, foreign key → families.id, ON DELETE CASCADE)
   - name (VARCHAR 255, not null)
   - icon (VARCHAR 50) -- emoji o icon name
   - created_by (UUID, foreign key → users.id)
   - created_at (TIMESTAMP, default NOW())

9. **shopping_items** - Item nelle liste
   - id (UUID, primary key)
   - list_id (UUID, foreign key → shopping_lists.id, ON DELETE CASCADE)
   - name (VARCHAR 255, not null)
   - quantity (VARCHAR 50) -- "2 kg", "1 confezione"
   - category (VARCHAR 50) -- 'fruits', 'vegetables', 'dairy', 'meat', 'pantry', etc.
   - note (TEXT)
   - checked (BOOLEAN, default false)
   - checked_by (UUID, foreign key → users.id)
   - checked_at (TIMESTAMP)
   - created_by (UUID, foreign key → users.id)
   - position (INTEGER) -- per ordinamento custom
   - created_at (TIMESTAMP, default NOW())

10. **shopping_history** - Storico acquisti (per AI)
    - id (UUID, primary key)
    - family_id (UUID, foreign key → families.id)
    - item_name (VARCHAR 255)
    - quantity (VARCHAR 50)
    - category (VARCHAR 50)
    - purchased_at (TIMESTAMP, default NOW())

11. **chore_templates** - Template faccende domestiche
    - id (UUID, primary key)
    - family_id (UUID, foreign key → families.id, ON DELETE CASCADE)
    - name (VARCHAR 255, not null)
    - description (TEXT)
    - estimated_minutes (INTEGER) -- durata stimata
    - difficulty (INTEGER) -- 1-5 per calcolo punti
    - recurrence_rule (TEXT) -- iCal RRULE
    - icon (VARCHAR 50)
    - created_at (TIMESTAMP, default NOW())

12. **chore_assignments** - Assegnazioni turni
    - id (UUID, primary key)
    - template_id (UUID, foreign key → chore_templates.id, ON DELETE CASCADE)
    - assigned_to (UUID, foreign key → users.id, ON DELETE CASCADE)
    - due_date (TIMESTAMP, not null)
    - completed (BOOLEAN, default false)
    - completed_at (TIMESTAMP)
    - skipped (BOOLEAN, default false)

13. **chore_points** - Punteggi gamification
    - id (UUID, primary key)

14. **ai_insights** - Insights generati da AI

    - title (VARCHAR 255)

    - dismissed (BOOLEAN, default false)
    - created_at (TIMESTAMP, default NOW())

````typescript
import { drizzle } from 'drizzle-orm/node-postgres';
import { migrate } from 'drizzle-orm/node-postgres/migrator';
import { Pool } from 'pg';
import * as dotenv from 'dotenv';

dotenv.config();

async function runMigrations() {
  const pool = new Pool({ connectionString: process.env.DATABASE_URL });
  const db = drizzle(pool);
  
  console.log('Running migrations...');
  await migrate(db, { migrationsFolder: './server/db/migrations' });
  console.log('Migrations completed!');
  
  await pool.end();
}

runMigrations().catch(console.error);
````

````typescript
import axios from 'axios';
import * as SecureStore from 'expo-secure-store';

const API_URL = process.env.EXPO_PUBLIC_API_URL || 'http://localhost:3000/api';

const api = axios.create({
  baseURL: API_URL,
  headers: {
    'Content-Type': 'application/json',
  },
});

// Request interceptor: aggiungi access token
api.interceptors.request.use(async (config) => {
  const token = await SecureStore.getItemAsync('accessToken');
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }
  return config;
});

// Response interceptor: handle 401 con refresh token
api.interceptors.response.use(
  (response) => response,
  async (error) => {
    if (error.response?.status === 401) {
      // Tenta refresh token
      const refreshToken = await SecureStore.getItemAsync('refreshToken');
      if (refreshToken) {
        try {
          const { data } = await axios.post(`${API_URL}/auth/refresh`, {
            refreshToken,
          });
          await SecureStore.setItemAsync('accessToken', data.accessToken);
          // Retry original request
          error.config.headers.Authorization = `Bearer ${data.accessToken}`;
          return axios(error.config);
        } catch {
          // Refresh failed, logout
          await SecureStore.deleteItemAsync('accessToken');
          await SecureStore.deleteItemAsync('refreshToken');
          // Navigate to login (implementare con navigation ref)
        }
      }
    }
    return Promise.reject(error);
  }
);

export default api;
````

````typescript
export interface SignupData {

  name: string;
}

  email: string;
  password: string;
}

export const authService = {
  async signup(data: SignupData) {
    const response = await api.post('/auth/signup', data);
    await this.saveTokens(response.data.accessToken, response.data.refreshToken);
    return response.data.user;
  },

  async login(data: LoginData) {
    const response = await api.post('/auth/login', data);
    await this.saveTokens(response.data.accessToken, response.data.refreshToken);
    return response.data.user;
  },

  async logout() {
    await SecureStore.deleteItemAsync('accessToken');
    await SecureStore.deleteItemAsync('refreshToken');
  },

  async getCurrentUser() {
    const response = await api.get('/auth/me');
    return response.data;
  },

  async saveTokens(accessToken: string, refreshToken: string) {
    await SecureStore.setItemAsync('accessToken', accessToken);
    await SecureStore.setItemAsync('refreshToken', refreshToken);
  },

  async isAuthenticated() {
    const token = await SecureStore.getItemAsync('accessToken');
    return !!token;
  },
};
````

````typescript
import { create } from 'zustand';
import { authService } from '../services/authService';

interface User {
  id: string;
  email: string;
  name: string;
  avatar_url?: string;
}

interface AuthState {
  user: User | null;
  isLoading: boolean;
  error: string | null;
  
  login: (email: string, password: string) => Promise<void>;
  signup: (email: string, password: string, name: string) => Promise<void>;
  logout: () => Promise<void>;
  loadUser: () => Promise<void>;
}

export const useAuthStore = create<AuthState>((set) => ({
  user: null,
  isLoading: false,
  error: null,

  login: async (email, password) => {
    set({ isLoading: true, error: null });
    try {
      const user = await authService.login({ email, password });
      set({ user, isLoading: false });
    } catch (error: any) {
      set({ error: error.message, isLoading: false });
      throw error;
    }
  },

  signup: async (email, password, name) => {
    set({ isLoading: true, error: null });
    try {
      const user = await authService.signup({ email, password, name });
      set({ user, isLoading: false });
    } catch (error: any) {
      set({ error: error.message, isLoading: false });
      throw error;
    }
  },

  logout: async () => {
    await authService.logout();
    set({ user: null });
  },

  loadUser: async () => {
    set({ isLoading: true });
    try {
      const user = await authService.getCurrentUser();
      set({ user, isLoading: false });
    } catch {
      set({ user: null, isLoading: false });
    }
  },
}));
````
